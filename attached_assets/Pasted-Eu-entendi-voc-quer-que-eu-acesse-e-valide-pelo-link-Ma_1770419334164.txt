import { Language, STORAGE_KEY } from "./types";

type Dict = Record<string, string>;
const dictionaries: Partial<Record<Language, Dict>> = {};

let currentLang: Language = "en";

function safeGetLocalStorage(key: string) {
  try {
    if (typeof window === "undefined") return null;
    return window.localStorage.getItem(key);
  } catch {
    return null;
  }
}

function safeSetLocalStorage(key: string, value: string) {
  try {
    if (typeof window === "undefined") return;
    window.localStorage.setItem(key, value);
  } catch {}
}

function detectDefaultLang(): Language {
  const saved = safeGetLocalStorage(STORAGE_KEY) as Language | null;
  if (saved) return saved;

  if (typeof navigator !== "undefined") {
    const nav = (navigator.language || "").toLowerCase();
    if (nav.startsWith("pt")) return "pt-BR";
    if (nav.startsWith("es")) return "es";
    if (nav.startsWith("fr")) return "fr";
  }
  return "en";
}

export function getLanguage(): Language {
  return currentLang;
}

export async function initI18n() {
  currentLang = detectDefaultLang();
  // carrega inglÃªs primeiro como fallback
  if (!dictionaries.en) {
    dictionaries.en = (await import("./locales/en.json")).default;
  }
  // carrega o idioma atual
  await loadLanguage(currentLang);
}

export async function loadLanguage(lang: Language) {
  currentLang = lang;
  safeSetLocalStorage(STORAGE_KEY, lang);

  if (!dictionaries.en) {
    dictionaries.en = (await import("./locales/en.json")).default;
  }
  if (!dictionaries[lang]) {
    dictionaries[lang] = (await import(`./locales/${lang}.json`)).default;
  }
}

export function t(key: string): string {
  const dict = dictionaries[currentLang] || {};
  const en = dictionaries.en || {};
  return dict[key] || en[key] || key;
}

client/src/i18n/useTranslation.ts

import { useEffect, useState } from "react";
import { getLanguage, initI18n, loadLanguage, t } from "./index";
import type { Language } from "./types";

export function useTranslation() {
  const [ready, setReady] = useState(false);
  const [lang, setLang] = useState<Language>(getLanguage());
  const [, rerender] = useState(0);

  useEffect(() => {
    (async () => {
      await initI18n();
      setLang(getLanguage());
      setReady(true);
      rerender(x => x + 1);
    })();
  }, []);

  async function changeLanguage(next: Language) {
    await loadLanguage(next);
    setLang(next);
    rerender(x => x + 1);
  }

  return { t, lang, ready, changeLanguage };
}
